---++Introduction

This is the Grid3 documented file from Marco. It needs to change slightly for OSG.

---++Information File Contents:

<pre>
#################################################################
#
# File: grid-info.conf
#
# Purpose: This file contains the configuration information
#          about the site.
#          It is used by the local MDS service and monitoring 
#          programs.
#
# Its path should be:
# ${GRID3_BASE_DIR}/monitoring/grid-info.conf
# With a simbolik link: 
# ln -s ${GRID3_BASE_DIR}/monitoring/grid3-info.conf ${GLOBUS_LOCATION}/etc/grid3-info.conf
#
# Placeholders starting with PH- have to be replaced if present (PH-HOSTNAME PH-VDTDIR).
# CHANGE-... require to be changed
#################################################################

##########################################################
#These values are modifiable by the administrator
GRID3_SITE_NAME="CHANGE-site-name"
#Site directories
GRID3_BASE_DIR="PH-VDTDIR"
GRID3_APP_DIR="/CHANGE-app"
GRID3_DATA_DIR="/CHANGE-data"
GRID3_TMP_DIR="/CHANGE-shared-tmp"
GRID3_TMP_WN_DIR="/CHANGE-tmp"

#These variables can contain a space separated list of values
#e.g. GRID3_JOB_CONTACT="grid01/jobmanager-condor grid01/jobmanager-pbs"
#Contacts for Job submission (JOB) and file transfers (UTIL)
GRID3_JOB_CONTACT="PH-HOSTNAME/jobmanager-condor"
GRID3_UTIL_CONTACT="PH-HOSTNAME/jobmanager"

#Please check to use the correct VO names
#The sponsors (1 or more) have to be names included in the list GRID3_VO_LIST (see below in this file)
#This variable can contain a space separated list of values
GRID3_SPONSOR=VO-NAME

#This variable can contain a space separated list of values
GRID3_SITE_INFO=""

##########################################################
#The following variables usually should not require changes

#List of grid3 VOs, to be substituted with a dynamic reading of filenames
GRID3_VO_LIST="btev gridex ivdgl ligo sdss usatlas uscms"

#File with mapping uname VO
#Please check the existence and the correctness of the account-VO mapping file
GRID3_USER_VO_MAP=${GRID3_BASE_DIR}/monitoring/grid3-user-vo-map.txt

#GridFTP (GSIftp) log file (specified with -Z option in /etc/xinetd.d/gsiftp )
#e.g.     server_args     = -l -a -G /your-GRID3_BASE_DIR/globus -Z /your-GRID3_BASE_DIR/globus/var/gridftp.log
#The suggested location is ${GRID3_BASE_DIR}/globus/var/gridftp.log or ${GLOBUS_LOCATION}/var/gridftp.log
GRID3_GRIDFTP_LOG=${GRID3_BASE_DIR}/globus/var/gridftp.log
#Used within the scripts, no export necessary
hostname="PH-HOSTNAME"

# Necessary to export the values
export GRID3_SITE_NAME
export GRID3_BASE_DIR
export GRID3_APP_DIR
export GRID3_DATA_DIR
export GRID3_TMP_DIR
export GRID3_TMP_WN_DIR
export GRID3_JOB_CONTACT
export GRID3_TRANSFER_CONTACT
export GRID3_SPONSOR
export GRID3_SITEINFO
# 
export GRID3_VO_LIST
export GRID3_USER_VO_MAP
export GRID3_GRIDFTP_LOG

</pre>
