---++ Installation
<verbatim>
source pacman-3.28/setup.sh
export VDT_GUMS_HOST=osp3.lbl.gov
cd osg-1.2-gsiftp/
pacman -get VTB:GridFTP
source setup.sh
vdt-control --list
vdt-post-install
vdt-control --list
$VDT_LOCATION/vdt/bin/vdt-ca-manage setupca --location local -url osg
cp $VDT_LOCATION/post-install/prima-authz.conf /etc/grid-security
cp $VDT_LOCATION/post-install/gsi-authz.conf /etc/grid-security
vi vdt/etc/vdt-local-setup.sh (to set opened ports)
source setup.sh
vdt-control --list
vdt-control --enable fetch-crl vdt-rotate-logs vdt-update-certs gsiftp gratia-gridftp-transfer gums-host-cron  vdt-control --on
$VDT_LOCATION/gums/scripts/gums-host-cron --gumsdebug
ls -l /opt/osg/osg-1.2-gsiftp/osg/etc/osg-user-vo-map.txt 
</verbatim>
